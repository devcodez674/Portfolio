CombatSystem.luaCombat
-- Combo Melee Combat System
local COMBO_WINDOW = 0.8
local HITBOX_SIZE  = Vector3.new(6,5,6)
local Combat = {}
Combat.__index = Combat
 
function Combat.new(char)
  return setmetatable({
    character=char, combo=0,
    lastHit=0, cooldown=false
  }, Combat)
end
 
function Combat:Attack()
  if self.cooldown then return end
  local now = tick()
  if now - self.lastHit > COMBO_WINDOW then
    self.combo = 0
  end
  self.combo += 1
  self.lastHit = now
  self:Hitbox(10 + self.combo * 3)
end
 
function Combat:Hitbox(dmg)
  local cf = self.character.HumanoidRootPart.CFrame
             * CFrame.new(0,0,-3)
  for _,p in workspace:GetPartBoundsInBox(cf,HITBOX_SIZE) do
    local h = p.Parent:FindFirstChild("Humanoid")
    if h and p.Parent ~= self.character then
      h:TakeDamage(dmg)
    end
  end
end
return Combat
⚔️ Dynamic combo damage with spatial hitboxes — smooth and expandable.

-- Secure Player DataStore Manager
local DSS   = game:GetService("DataStoreService")
local Plrs  = game:GetService("Players")
local Store = DSS:GetDataStore("PlayerData_v3")
local cache = {}
 
local DEFAULT = {
  coins=0, level=1, xp=0,
  inventory={},
  stats={kills=0,deaths=0}
}
 
local function deepCopy(t)
  local c={}
  for k,v in pairs(t) do
    c[k]=type(v)=="table" and deepCopy(v) or v
  end; return c
end
 
local function Load(p)
  local ok,d=pcall(function()
    return Store:GetAsync("u_"..p.UserId)
  end)
  if ok then
    cache[p.UserId]=d or deepCopy(DEFAULT)
  else
    p:Kick("Data load failed.")
  end
end
 
local function Save(p)
  local d=cache[p.UserId]
  if d then
    pcall(function()
      Store:SetAsync("u_"..p.UserId,d)
    end)
    cache[p.UserId]=nil
  end
end
 
Plrs.PlayerAdded:Connect(Load)
Plrs.PlayerRemoving:Connect(Save)
💾 pcall-protected saves with deep copy defaults — data that never corrupts.
EnemyAI.luaAI
-- Patrol / Chase / Attack AI
local PFS   = game:GetService("PathfindingService")
local AGGRO = 30
local AI = {}
AI.__index = AI
 
function AI.new(npc,wps)
  return setmetatable({
    npc=npc, waypoints=wps,
    state="patrol", target=nil
  }, AI)
end
 
function AI:Scan()
  local pos=self.npc.HumanoidRootPart.Position
  for _,p in game.Players:GetPlayers() do
    local c=p.Character
    if c then
      local d=(pos-c.HumanoidRootPart.Position).Magnitude
      if d<=AGGRO then return c end
    end
  end
end
 
function AI:Chase(tgt)
  local path=PFS:CreatePath({AgentRadius=2})
  path:ComputeAsync(
    self.npc.HumanoidRootPart.Position,
    tgt.HumanoidRootPart.Position
  )
  for _,wp in path:GetWaypoints() do
    self.npc.Humanoid:MoveTo(wp.Position)
    self.npc.Humanoid.MoveToFinished:Wait()
  end
end
 
function AI:Tick()
  self.target=self:Scan()
  if self.target then
    self.state="chase"
    self:Chase(self.target)
  else
    self.state="patrol"
  end
end
return AI
🤖 State machine — idles, patrols, detects players, chases with real pathfinding.
UIController.luaUI / GUI
-- Animated Notification + HUD System
local TS  = game:GetService("TweenService")
local plr = game.Players.LocalPlayer
local IN  = TweenInfo.new(.35,Enum.EasingStyle.Back)
local OUT = TweenInfo.new(.25,Enum.EasingStyle.Quad)
local UI  = {}
 
function UI.Notify(msg,color,dur)
  local gui=plr.PlayerGui:FindFirstChild("Notify")
  if not gui then return end
  local n=gui.Template:Clone()
  n.Label.Text=msg
  n.Frame.BackgroundColor3=color
  n.Parent=gui.Stack
  n.Position=UDim2.new(1.2,0,0,0)
  TS:Create(n,IN,{
    Position=UDim2.new(0,0,0,0)
  }):Play()
  task.delay(dur or 3,function()
    local t=TS:Create(n,OUT,{
      Position=UDim2.new(-1.2,0,0,0)
    })
    t:Play()
    t.Completed:Connect(function()
      n:Destroy()
    end)
  end)
end
 
function UI.SetHealth(hp,max)
  local bar=plr.PlayerGui.HUD.HealthBar
  TS:Create(bar,IN,{
    Size=UDim2.new(hp/max,0,1,0)
  }):Play()
end
return UI
🖥️ Slide-in/out notifications with Back easing + live health bar tweening.
